option( CASTOR_BUILD_PLUGINS "Build Castor3D plugins" TRUE )

if ( ${CASTOR_BUILD_PLUGINS} )
	add_subdirectory( Renderers )
	add_subdirectory( Importers )
	add_subdirectory( Dividers )
	add_subdirectory( Techniques )
	add_subdirectory( ToneMappings )
	add_subdirectory( PostEffects )
	add_subdirectory( Generic )
endif ()

macro( PluginsInit )
	if ( ${CASTOR_BUILD_PLUGINS} )
		RenderersInit()
		ImportersInit()
		DividersInit()
		TechniquesInit()
		ToneMappingsInit()
		PostEffectsInit()
		GenericInit()
	endif ()
endmacro()

macro( PluginsBuild )
	if ( ${CASTOR_BUILD_PLUGINS} )
		RenderersBuild()
		ImportersBuild()
		DividersBuild()
		TechniquesBuild()
		ToneMappingsBuild()
		PostEffectsBuild()
		GenericBuild()
	endif ()
endmacro()

macro( PluginsSummary )
	if ( ${CASTOR_BUILD_PLUGINS} )
		RenderersSummary( ${msg} ${msgtest} )
		ImportersSummary( ${msg} ${msgtest} )
		DividersSummary( ${msg} ${msgtest} )
		TechniquesSummary( ${msg} ${msgtest} )
		ToneMappingsSummary( ${msg} ${msgtest} )
		PostEffectsSummary( ${msg} ${msgtest} )
		GenericSummary( ${msg} ${msgtest} )
	endif ()
endmacro()

macro( PluginsSetup )
	if ( ${CASTOR_BUILD_PLUGINS} )
		RenderersSetup()
		ImportersSetup()
		DividersSetup()
		TechniquesSetup()
		ToneMappingsSetup()
		PostEffectsSetup()
		GenericSetup()
	endif ()
endmacro()
